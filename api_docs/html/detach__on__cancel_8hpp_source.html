103template <typename UpstreamSender, typename DownstreamReceiver>
104struct operation_state<UpstreamSender, DownstreamReceiver>::detached_state final {
105 using parent_op_t =
106 typename operation_state<UpstreamSender, DownstreamReceiver>::type;
107
108 detached_state(
109 parent_op_t& op,
110 UpstreamSender&& s) //
111 noexcept(is_nothrow_connectable_v<UpstreamSender, _receiver>)
112 : parentOp_(init_refcount(op))
113 , childOp_(connect(std::move(s), _receiver{this})) {}
114
115 static constexpr std::uintptr_t mask = ~std::uintptr_t{3u};
116 // operation_state::type* stored as integer: low 2 bits for ref count
117 std::atomic_uintptr_t parentOp_;
118 inplace_stop_source stopSource_;
119 connect_result_t<UpstreamSender, _receiver> childOp_;
120
121 void request_stop() noexcept {
122 auto expected = parentOp_.load(std::memory_order_relaxed);
123 if (ref_count(expected) == 0u) {
124 // try_get_op already executed
125 return;
126 }
127 UNIFEX_ASSERT(ref_count(expected) == 1u);
128 // 1. set ref count to 2
129 // 2. zero the pointer
130 // if successful, callback owns the op and ptr remains in `expected`
131 if (!parentOp_.compare_exchange_strong(
132 expected,
133 2u,
134 std::memory_order_acq_rel,
135 std::memory_order_relaxed)) {
136 UNIFEX_ASSERT(ref_count(expected) == 0u);
137 return;
138 }
139 stopSource_.request_stop();
140 auto refCount = parentOp_.fetch_sub(1u, std::memory_order_acq_rel);
141 UNIFEX_ASSERT(parent_op_ptr(refCount) == nullptr);
142 auto op = parent_op_ptr(expected);
143 op->callback_.destruct();
144 if (refCount == 1u) {
145 // callback owns the detached state
146 op->state_.reset();
147 } else {
148 (void)op->state_.release();
149 }
150 unifex::set_done(std::move(op->receiver_));
151 }
152
153 parent_op_t* try_get_op() noexcept {
154 auto op = parentOp_.fetch_sub(1u, std::memory_order_acq_rel);
155 if (ref_count(op) != 1u) {
156 // decrement from 2 means lost race with stop callback
157 UNIFEX_ASSERT(ref_count(op) == 2u);
158 UNIFEX_ASSERT(parent_op_ptr(op) == nullptr);
159 return nullptr;
160 }
161 if (auto ptr = parent_op_ptr(op)) {
162 ptr->callback_.destruct();
163 return ptr;
164 }
165 // stop owns callback destruction
166 delete this;
167 return nullptr;
168 }
169
170private:
171 // ref count in low 2 bits: start at 1
172 static std::uintptr_t init_refcount(parent_op_t& parentOp) noexcept {
173 return reinterpret_cast<std::uintptr_t>(&parentOp) | std::uintptr_t{1u};
174 }
175
176 static parent_op_t* parent_op_ptr(std::uintptr_t parentOp) noexcept {
177 return reinterpret_cast<parent_op_t*>(parentOp & mask);
178 }
179
180 static std::uintptr_t ref_count(std::uintptr_t parentOp) noexcept {
181 return parentOp & ~mask;
182 }
183};
