364 void source_cleanup_done() noexcept {
365 if (!cleanupCompleted_.load(std::memory_order_acquire)) {
366 if (!cleanupCompleted_.exchange(true, std::memory_order_acq_rel)) {

368 // Let the other operation call the final receiver.
369 return;

456 void trigger_next_done() noexcept {
457 if (!cleanupReady_.load(std::memory_order_acquire)) {
458 stopSource_.request_stop();
459 if (!cleanupReady_.exchange(true, std::memory_order_acq_rel)) {
460 // Successfully registered completion of next(trigger)
461 // before someone called cleanup(stream). We have passed
462 // responsibility for calling cleanup(trigger_) to the
463 // call to start() on the cleanup(stream) sender.
464 return;
465 }
466 }
467
468 // Otherwise, the cleanup(stream) operation has already been started
469 // before the next(trigger) operation finished.
470 // We have the responsibility for launching cleanup(trigger).
471 UNIFEX_ASSERT(cleanupOperation_ != nullptr);
472 cleanupOperation_->start_trigger_cleanup();
