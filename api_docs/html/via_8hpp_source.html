36constexpr blocking_kind _blocking_kind(blocking_kind predBlocking, blocking_kind succBlocking) noexcept {
37 if (predBlocking == blocking_kind::never &&
38 succBlocking == blocking_kind::never) {
39 return blocking_kind::never;
40 } else if (
41 predBlocking == blocking_kind::always_inline &&
42 succBlocking == blocking_kind::always_inline) {
43 return blocking_kind::always_inline;
44 } else if (
45 (predBlocking == blocking_kind::always_inline ||
46 predBlocking == blocking_kind::always) &&
47 (succBlocking == blocking_kind::always_inline ||
48 succBlocking == blocking_kind::always)) {
49 return blocking_kind::always;
50 } else {
51 return blocking_kind::maybe;
52 }
53}
