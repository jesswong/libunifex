99 auto stopToken) noexcept {
100 return unifex::let_value_with(
101 [stopToken, evt]() noexcept {
102 // unifex::async_manual_reset_event::async_wait() returns an
103 // unstoppable *Sender* so, to support prompt cancellation
104 // of the *Sender* we're composing here, we register a stop
105 // callback with our *Receiver* that responds to stop
106 // requests by transitioning the auto-reset event to the
107 // done state. When the auto-reset event transitions to the
108 // done state, anyone waiting on our next-sender will wake
109 // up and complete with set_done.
110 //
111 // This strategy matches the *Stream* contract because
112 // cancelling the result of next(stream) is expected to
113 // cancel the whole stream. The streams returned from
114 // event.stream() are light weight handles to the overall
115 // event and are indistinguishable from each other so
116 // cancelling one such stream by cancelling its next-sender
117 // is interpreted as cancelling the whole event.
118 auto stopCallback = [evt]() noexcept {
119 evt->set_done();
120 };
121
122 // avoid inline `Scheduler`: narrow margin for a lifetime issue
123 // during synchronous cancellation
124 //
125 // 1. when cancelling on the same `Scheduler` that's waiting on
126 // `async_wait()`: the stop callback never synchronously wakes the
127 // event because `evt->set_done()` always schedules the wakeup
128 //
129 // 2. when cancelling from a different `Scheduler` than the one
130 // that's waiting on `async_wait()`: there's a race between
131 // completing the stop callback and completing the `async_wait()`
132 // but the `let_value_with` operation state will not complete the
133 // downstream `Receiver` until the stop callback has been destroyed,
134 // which will synchronize with the completion of the callback; given
135 // this scenario is "cancel and complete on different Schedulers",
136 // the synchronization will block rather than no-op
137 using stop_token_t = unifex::remove_cvref_t<decltype(stopToken)>;
138 using stop_callback_t =
139 typename stop_token_t::template callback_type<
140 decltype(stopCallback)>;
141
142 return stop_callback_t{stopToken, stopCallback};
143 },
144 [evt](auto&) noexcept {
145 return unifex::let_value(
146 evt->event_.async_wait(), [evt]() noexcept {
147 return unifex::just_void_or_done(evt->try_reset());
148 });
149 });
150 });
