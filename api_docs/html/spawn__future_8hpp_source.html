43// Spawning a future creates a race condition between the spawned operation
44// completing and the future trying to observe the result; we resolve this race
45// by moving the operation through a state machine that reflects what work has
46// been done and who's waiting for whom.
47//
48// Successfully-spawned operations start in the init state; from there, the
49// simplest path is for the operation to complete normally before the future's
50// continuation consumes the result, in which case the operation moves from init
51// to one of value, error, or done (depending on the result) and then the
52// future deletes the spawned operation as a side effect of completing.
53//
54// There are several other paths through the state machine:
55//
56// 1. The future is dropped (i.e. destroyed beforing being connected, or
57// destroyed after connect but before start) before the spawned operation
58// completes. In this scenario, we request stop on the spawned operation and
59// then move to the complete state; when the operation ultimately completes,
60// it observes the complete state and cleans up.
61//
62// 2. The future is dropped after the spawned operation completes. In this
63// scenario, we'll observe that the operation is in the value, error, or done
64// state and clean up as a side effect of dropping the future.
65//
66// 3. The future is connected and started, but then receives a stop request
67// before the operation completes. In this scenario, we need to signal the
68// operation and then complete the future with set_done(). To do this, we
69// request stop on the spawned operation and then move the operation to the
70// abandoned state; as the future completes with done in this scenario, we
71// need to negotiate who will delete the spawned operation: the now-completed
72// future or the spawned operation? We resolve this negotiation by having
73// both sides race to see who can move the state from abandoned to complete;
74// whoever *fails* to update the state is deemed responsible for cleaning up.
75//
76// 4. The future is connected and started, but then receives a stop request
77// after the operation completes. In this final scenario, the stop request
78// is effectively ignored. The operation will move the state from init to
79// one of value, error, or done, and then the future will complete with the
80// appropriate result as if the stop request was never received.
81//
82// All of the above-described state transitions happen with atomic updates to
83// the operation's state_ field so any "happens-before" relationships are as
84// observed by those atomic operations.
85enum class _future_state : unsigned char {
86 // the operation has been constructed, and perhaps started
87 init,
88 // the future received a stop request and managed to signal so before the
89 // operation completed
90 abandoned,
91 // the operation completed with set_value(); the values_ member has been
92 // constructed
93 value,
94 // the operation completed with set_error(); the error_ member has been
95 // constructed
96 error,
97 // the operation completed with set_done()
98 done,
99 // one of three things happened: a) the future was dropped, b) the future was
100 // cancelled and it completed with done before the spawned operation
101 // completed, or c) the future was cancelled and the spawned operation
102 // completed before the future could complete with done; regardless, whoever
103 // (future or operation) observes the complete state is responsible for
104 // deleting the operation
105 complete,
106};

141 // invoked when the waiting future receives a stop request; attempts to cancel
142 // the spawned operation and ensure that the waiting future is awoken promptly
143 void abandon() noexcept {
144 // abandon() is only invoked from a stop callback that's registered when the
145 // future is awaited; we know for sure that the future has been started and
146 // will consume the spawned operation. if abandonment fails, it means the
147 // stop request has lost the race to a natural completion and the future
148 // will produce whatever the operation produced.
149
150 // the operation must either be in the init state or one of the natural
151 // completion states; in the former case we want to mark it as abandoned and
152 // in the latter we'll just allow the future to complete naturally
153 auto expected = _future_state::init;
154 if (state_.compare_exchange_strong(
155 expected,
156 _future_state::abandoned,
157 // on success, there are two audiences to publish to: the waiting
158 // future and the still-running operation; the future will
159 // synchronize through evt_ and see everything we've written; the
160 // still running operation will eventually complete and have to
161 // negotiate completion with the completing future so relaxed is
162 // fine at this point.
163 //
164 // on failure, we've lost a completion race with the spawned
165 // operation, in which case the state is a completion signal and the
166 // operation is waking up the future with evt_.set(), which causes a
167 // synchronization between operation and future; the attempted
168 // abandonment "never happened" and so there's nothing to
169 // synchronize
170 std::memory_order_relaxed)) {
171 // we set the state to abandoned; the future will complete with set_done()
172
173 stopSource_.request_stop();
174
175 // publish the result
176 evt_.set();
177 } else {
178 UNIFEX_ASSERT(
179 expected == _future_state::value ||
180 expected == _future_state::error || expected == _future_state::done);
181 }
182 }

184 // invoked by the spawned operation when it completes; attempts to record the
185 // operation's completion style (value, error, or done) and store the result
186 // in case of value or error.
187 template <typename Func>
188 void complete(_future_state desired, Func func) noexcept {
189 UNIFEX_ASSERT(
190 desired == _future_state::value || desired == _future_state::error ||
191 desired == _future_state::done);
192
193 // the happy path is that we transition from init to the desired state
194 auto expected = _future_state::init;
195 if (state_.compare_exchange_strong(
196 expected,
197 desired,
198 // on success, we'll invoke evt_.set() to publish the writes that
199 // happen in func() and destruct_op(); if the future is ever
200 // started, it will synchronize through evt_ to consume the result;
201 // if the future is never started then drop() will observe that the
202 // future was dropped after the spawned operation completed and
203 // synchronize through evt_.ready() before deleting this operation
204 // state
205 std::memory_order_relaxed,
206 // on failure we need to consume the future's writes in case
207 // negotiate_deletion() decides the next step is to delete this
208 // operation state
209 std::memory_order_acquire)) {
210 // we successfully set the state to a completion signal (value, error, or
211 // done) so invoke func to store the operation's result
212 func();
213
214 // func() has stored whatever it needs to in either values_ or error_ so
215 // we can safely destruct the nested operation state now. we have to do
216 // this *before* waking up the future because the future might wake up on
217 // another thread and delete the whole spawned operation while we're busy
218 // trying to destroy a sub-part. we also need to do this first because
219 // we're relying on the event's memory ordering semantics to publish our
220 // effects to the future and, since we've succeeded in completing here,
221 // the future is going to be the one to destroy the overall operation,
222 // which it should do *after* the nested operation has been destructed.
223 destruct_op();
224
225 // now that the spawned operation is really done, wake up the future and
226 // let it take things from here
227 evt_.set();
228 } else {
229 // the future has disappeared before we completed so we need to coordinate
230 // deletion of this operation state
231
232 // If expected is abandoned then the future is currently responding to a
233 // stop request; if expected is complete then either the future was
234 // dropped before being started or it was cancelled and has already
235 // completed. There's no other valid state here.
236 UNIFEX_ASSERT(
237 expected == _future_state::abandoned ||
238 expected == _future_state::complete);
239
240 // Figure out which case we're in and delete the operation state; do this
241 // in a helper function because it's not dependent on Func
242 negotiate_deletion(expected);
243 }
244 }
245
246 void negotiate_deletion(_future_state expected) noexcept {
247 // the spawned operation has completed so we can unconditionally destruct it
248 destruct_op();
249
250 if (expected == _future_state::abandoned) {
251 // the future abandoned the operation but hasn't dropped its ownership
252 // stake, yet; we need to coordinate who's going to delete the operation
253 // state
254 if (state_.compare_exchange_strong(
255 expected,
256 _future_state::complete,
257 // on success we need to publish our writes to the future's thread
258 std::memory_order_release,
259 // on failure we need to observe the future's writes
260 std::memory_order_acquire)) {
261 // we've handed ownership of the operation state to the future so just
262 // bail out
263 return;
264 }
265 }
266
267 // the future has disappeared and we own clean-up; it could be that the
268 // future was already gone when we tried to set state_ to a completion
269 // signal, or it could be that we observed the abandoned state and then lost
270 // the race to set state_ to complete
271 UNIFEX_ASSERT(expected == _future_state::complete);
272
273 // we own deletion
274 deleter_(this, expected);
275 }

277 // invoked by the future if it's dropped before being started
278 void drop() noexcept {
279 auto state = state_.load(
280 // either we'll see init, in which case we'll do more synchronizing, or
281 // we'll see a completion signal, in which case we'll synchronize
282 // through evt_
283 std::memory_order_relaxed);
284
285 switch (state) {
286 case _future_state::init:
287 // we're being dropped before the spawned operation has completed;
288 // request stop to hurry it up
289 stopSource_.request_stop();
290
291 // we want to give the spawned operation responsibility to delete the
292 // operation state but it might have finished since we read the state as
293 // init so try to assign the complete state with a CAS
294 if (state_.compare_exchange_strong(
295 state,
296 _future_state::complete,
297 // on success we need to publish our writes to the spawned
298 // operation
299 std::memory_order_release,
300 // on failure we'll synchronize the operation's final write by
301 // reading evt_
302 std::memory_order_relaxed)) {
303 // we've assigned clean-up responsibility to the not-yet-completed
304 // operation; we're done
305
306 return;
307 }
308
309 // we lost the race; this means the operation completed and we're
310 // responsible for clean-up
311 UNIFEX_ASSERT(
312 state == _future_state::value || state == _future_state::error ||
313 state == _future_state::done);
314
315 [[fallthrough]];
316
317 case _future_state::value:
318 case _future_state::error:
319 case _future_state::done:
320 // we're being dropped after the operation finished; we need to clean up
321 // the values_ or error_ that was written by the spawned operation
322 // because the future isn't going to consume them naturally
323
324 // reading evt_.ready() performs a load-acquire on the event so this is
325 // how we consume the operation's last writes
326 (void)evt_.ready();
327
328 // having synchronized with evt_, we can now clean up
329 deleter_(this, state);
330
331 return;
332
333 default: // should never happen
334 std::terminate();
335 }
336 }

340 template <typename Alloc>
341 Alloc get_allocator() const noexcept {
342 using byte_alloc_t =
343 typename std::allocator_traits<Alloc>::template rebind_alloc<std::byte>;
344 using op_alloc_t = typename _spawn_future_op_alloc<byte_alloc_t>::type;
345
346 // see the class-level comment for _spawn_future_op_impl<>::type that
347 // explains that type's memory layout
348 //
349 // we know that this points into the "middle" of the spawned operation,
350 // right after the _spawn_future_op_alloc sub-object, so we can move
351 // backwards from this by the size of a _spawn_future_op_alloc<Alloc> to get
352 // to our allocator
353 const std::byte* bytePtr = reinterpret_cast<const std::byte*>(this);
354
355 bytePtr -= sizeof(op_alloc_t);
356
357 const op_alloc_t* op = reinterpret_cast<const op_alloc_t*>(bytePtr);
358
359 return op->alloc_;
360 }

491 template <typename... Values>
492 void set_value(Values&&... values) noexcept {
493 op()->complete(_future_state::value, [&, op = op()]() noexcept {
494 // we set the state to value
495 UNIFEX_TRY {
496 activate_union_member(op->values_, static_cast<Values&&>(values)...);
497 }
498 UNIFEX_CATCH(...) {
499 // we failed to store the value so update the state and store the
500 // current exception
501 op->state_.store(
502 _future_state::error,
503 // since we made it here, the curent value of state_ is value, which
504 // was stored with release semantics inside complete()
505 //
506 // updating from value to error with relaxed semantics is fine
507 // because the write to op->error_ will be published when complete()
508 // sets evt_ after we return
509 std::memory_order_relaxed);
510 activate_union_member(op->error_, std::current_exception());
511 }
512 });
513 }

517 // we set the state to error

519 });
520 }

633 static void
634 deleter(_spawn_future_op_base* base, _future_state state) noexcept {
635 // get an allocator for *this* type
636 using alloc_t =
637 typename std::allocator_traits<Alloc>::template rebind_alloc<type>;
638
639 auto self = static_cast<type*>(base);
640 alloc_t alloc = self->alloc_;
641
642 // state is a function argument to avoid reloading the current state from
643 // the state_ member when we already know its current value
644 UNIFEX_ASSERT(state == self->state_.load(std::memory_order_relaxed));
645
646 if (state == _future_state::value) {
647 deactivate_union_member(self->values_);
648 } else if (state == _future_state::error) {
649 deactivate_union_member(self->error_);
650 }
651
652 std::allocator_traits<alloc_t>::destroy(alloc, self);
653 std::allocator_traits<alloc_t>::deallocate(alloc, self, 1);
654 }

701// A callable that returns a Sender given an inplace_stop_token. An instance of
702// this type is given to let_value_with_stop_token while constructing a
703// future<>.
704template <typename... T>
705struct _future_sender_from_stop_token<T...>::type final {
706 using spawn_future_op_t = typename _spawn_future_op<T...>::type;
707
708 // an owning handle type that drops a spawned operation upon destruction
709 using op_handle = std::unique_ptr<spawn_future_op_t, _op_dropper>;
710
711 op_handle op_;
712
713 explicit type(spawn_future_op_t* op) noexcept : op_(op) {}
714
715 auto operator()(inplace_stop_token stopToken) noexcept {
716 return let_value_with(
717 _future_stop_callback_factory{op_.get(), stopToken},
718 [this](auto&) noexcept {
719 return let_value(
720 op_->evt_.async_wait(),
721 [this]() noexcept(
722 noexcept(op_->get_value_sender(), op_->get_error_sender())) {
723 auto rawOp = op_.release();
724
725 using value_t = decltype(op_->get_value_sender());
726 using error_t = decltype(op_->get_error_sender());
727 using done_t = decltype(just_done());
728
729 using return_t = variant_sender<value_t, error_t, done_t>;
730
731 auto state = rawOp->state_.load(std::memory_order_relaxed);
732
733 // we capture state by reference because it may be updated by
734 // the compare_exchange_strong below
735 scope_guard cleanup = [rawOp, &state]() noexcept {
736 rawOp->deleter_(rawOp, state);
737 };
738
739 if (state == _future_state::abandoned) {
740 if (rawOp->state_.compare_exchange_strong(
741 state,
742 _future_state::complete,
743 // on success, publish our writes to the still-running
744 // spawned operation
745 std::memory_order_release,
746 // on failure, consume the now-finished operation's
747 // writes
748 std::memory_order_acquire)) {
749 // we gave clean-up responsibility to the spawned operation
750 cleanup.release();
751 } else {
752 // we run this if the CAS failed, which should mean the
753 // spawned operation beat us to setting state to complete
754 UNIFEX_ASSERT(state == _future_state::complete);
755 }
756 }
757
758 switch (state) {
759 case _future_state::value:
760 return return_t{rawOp->get_value_sender()};
761
762 case _future_state::error:
763 return return_t{rawOp->get_error_sender()};
764
765 case _future_state::done:
766 case _future_state::abandoned:
767 case _future_state::complete: return return_t{just_done()};
768
769 default:
770 // doesn't happen
771 std::terminate();
772 }
773 });
774 });
775 }
776};

778// A future<Scope, T...> is a Sender of T... that, when awaited, completes with
779// the result of an operation that was earlier spawned in an async_scope of type
780// Scope.
781//
782// A future<> is nested within its associated scope, which means that a) it may
783// have *failed* to nest, in which case there is no associated spawned operation
784// and the future<> can only complete with set_done(), and b) if it was
785// successfully nested then the future<> holds a reference on its associated
786// scope until it is either discarded or completed.
787template <typename Scope, typename... T>
788struct [[nodiscard]] _future<Scope, T...>::type final {
789private:
790 using spawn_future_op_t = typename _spawn_future_op<T...>::type;

792 // constructs the Sender
793 static auto make_sender(spawn_future_op_t* op) noexcept {
794 using callable = typename _future_sender_from_stop_token<T...>::type;
795
796 return let_value_with_stop_token(callable{op});
797 }

799 using sender_t = decltype(nest(make_sender(nullptr), UNIFEX_DECLVAL(Scope&)));
800
801 sender_t sender_;
802
803 // this is just good hygiene but let's assert it because we depend on it for
804 // *our* move operators to be nothrow
805 static_assert(std::is_nothrow_move_constructible_v<sender_t>);
806 static_assert(std::is_nothrow_move_assignable_v<sender_t>);
807
808public:
809 template <
810 template <typename...>
811 class Variant,
812 template <typename...>
813 class Tuple>
814 using value_types = Variant<Tuple<T...>>;
815
816 template <template <typename...> class Variant>
817 using error_types = Variant<std::exception_ptr>;
818
819 static constexpr bool sends_done = true;
820
821 explicit type(
822 Scope& scope,
823 typename _spawn_future_op<T...>::type*
824 op) noexcept(noexcept(nest(make_sender(op), scope)))
825 : sender_(nest(make_sender(op), scope)) {}

827 type(type&&) noexcept = default;
828
829 ~type() = default;
830
831 type& operator=(type&&) noexcept = default;
832

836 is_nothrow_connectable_v<sender_t, Receiver>) {
837 return connect(std::move(self).sender_, static_cast<Receiver&&>(receiver));

846template <typename Sender, typename Scope, typename Alloc>
847using future_for = typename _spawn_future_op_impl<
848 Sender,
849 Scope,
850 typename std::allocator_traits<Alloc>::template rebind_alloc<std::byte>>::
851 type::future_t;
852
853struct _spawn_future_fn {
854private:
855 struct deref;
856
857public:
858 template(
859 typename Sender,
860 typename Scope,
861 typename Alloc = std::allocator<std::byte>) //
862 (requires typed_sender<remove_cvref_t<Sender>> AND
863 std::is_invocable_v<tag_t<nest>, Sender, Scope&> AND
864 is_allocator_v<Alloc>) //
865 auto
866 operator()(Sender&& sender, Scope& scope, const Alloc& alloc = {}) const {
867 // We need to do several things here and the ordering is nuanced:
868 // - We should provide the Strong Exception Guarantee: if an exception is
869 // thrown then the state of the programme should be rolled back to the
870 // state just before we were called; this means no memory leaks, no
871 // running operations, and no leaked references to the given scope. There
872 // are comments interwoven with the implementation that explain how we
873 // meet this requirement.
874 // - We'll be nesting two Senders in the given scope (the given Sender and
875 // the future that we return). We must tolerate the scope being
876 // closed a) before we start, b) between nesting the two Senders, and
877 // c) after we've nested both Senders. Handling c) is trivial; a) is
878 // not hard; b) is tricky.
879 //
880 // Considering the ordering of nesting Senders, there are two options:
881 // 1. If we nest the spawned operation before nesting the future then we
882 // risk the former succeeding and the latter failing, which would
883 // leave us with a running operation whose result can't be observed
884 // (because a failed nest() is analogous to just_done()--it will
885 // complete synchronously with set_done() once observed).
886 // 2. If we nest the future before nesting the operation then we avoid the
887 // above problem; if the former succeeds and the latter fails then
888 // we have a successfully-nested future that, when awaited, will
889 // observe the failed operation as having synchronously completed
890 // with set_done().
891 //
892 // We choose option 2 to ensure we don't waste time running an unobservable
893 // operation. Obviously, if both nest operations succeed, the future can be
894 // used to observe the result of the spawned operation. If both nest
895 // operations fail then we'll have allocated an operation state in which the
896 // spawned operation completes with set_done() and we'll have constructed a
897 // future that will complete with set_done() without observing the allocated
898 // operation state.
899 //
900 // The obvious downside to this algorithm is that we sometimes make
901 // allocations that could, perhaps, be elided. In the case that nesting
902 // the spawned operation fails, we allocate space for an operation that,
903 // in principle, we know will complete with set_done() so it seems like
904 // we could skip the allocation and just mark the future as cancelled;
905 // given that the future is nested successfully, though, the current
906 // implementation unifies successful and unsuccessful nesting of the
907 // spawned operation: a successfully nested future always consults the
908 // spawned operation state for its completion. In the case that both
909 // nests fail, the heap allocation seems particularly useless because
910 // the future won't even consult it for its result. It's not obvious to
911 // me how to detect that a nest operation has failed so I don't know how
912 // to elide unnecessary allocations. Perhaps we could elide the
913 // allocation if there were a "reserve" operation on an async_scope.
914
915 // rebind Alloc to be a std::byte allocator; this ensures that
916 // _spawn_future_op_impl is always instantiated with a std::byte allocator,
917 // cutting down on a potential explosion of template instantiations.
918 using byte_alloc_t =
919 typename std::allocator_traits<Alloc>::template rebind_alloc<std::byte>;
920
921 using op_t = typename _spawn_future_op_impl<
922 remove_cvref_t<Sender>,
923 Scope,
924 byte_alloc_t>::type;
925
926 // rebind again because we need to allocate op_ts, not std::bytes; we need
927 // both an allocator and its std::allocator_traits
928 using alloc_traits_t =
929 typename std::allocator_traits<Alloc>::template rebind_traits<op_t>;
930 using op_alloc_t = typename alloc_traits_t::allocator_type;
931
932 // now actually construct the necessary allocator; the spec says this has to
933 // be a noexcept expression for Alloc to conform to the Allocator concept
934 // but, since this is the first thing we're doing, if it throws in practice
935 // we don't care--we can just let the exception escape
936 op_alloc_t opAlloc{std::move(alloc)};
937
938 // allocate space for the spawned operation; we expect this to throw if the
939 // allocation fails but, again, we don't care--letting the exception escape
940 // is the right thing to do
941 auto* op = alloc_traits_t::allocate(opAlloc, 1);
942
943 // now construct the spawned operation in the just-allocated space; we can't
944 // assert that this is noexcept because
945 // std::allocator_traits<T>::construct() is declared to throw rather than
946 // being conditionally-noexcept, but we know that the underlying constructor
947 // is noexcept so we'll assume this doesn't throw in practice
948 alloc_traits_t::construct(opAlloc, op, opAlloc);
949
950 // the next two steps might throw and we have to destroy op if one of them
951 // does throw so set up a scope_guard that will do that for us;
952 // conveniently, the deleter() static method does exactly the right thing
953 scope_guard cleanUp = [op]() noexcept {
954 // Constructing the future is *almost* no-throw--only the call to nest()
955 // might throw--so the future will invoke drop() on the operation (moving
956 // it from init to complete) as part of its destructor before this code
957 // runs.
958 op_t::deleter(op, _future_state::complete);
959 };
960
961 using future_t = future_for<remove_cvref_t<Sender>, Scope, byte_alloc_t>;
962
963 // construct the future that we'll hand back to the caller; this is fairly
964 // likely to be noexcept since it's not much more than a Sender construction
965 // but it depends on the implementation of nest() for scope so it might
966 // throw
967 future_t future{scope, op};
968
969 // now finally construct the spawned operation, which might throw
970 op->init_operation(static_cast<Sender&&>(sender), scope, op);
971
972 // now that everything is wired together there are no more exception
973 // concerns
974 cleanUp.release();
975
976 // start the spawned operation
977 start(*op);
978
979 // ideally, the compiler performs NRVO and constructs this future in-place
980 // at the call site but future<>'s move constructor is noexcept so even if
981 // there's an actual move here, there's still no chance of an exception
982 return future;
983 }

987 noexcept(
988 is_nothrow_callable_v<tag_t<bind_back>, deref, Scope*, const Alloc&>)
989 -> std::enable_if_t<
990 is_allocator_v<Alloc>,
991 bind_back_result_t<deref, Scope*, const Alloc&>>;
992};
