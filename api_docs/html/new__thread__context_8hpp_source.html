155template <typename Receiver>
156inline void _op<Receiver>::type::start() & noexcept {
157 UNIFEX_TRY {
158 // Acquire the lock before launching the thread.
159 // This prevents the run() method from trying to read the thread_ variable
160 // until after we have finished assigning it.
161 //
162 // Note that this thread_ variable is private to this particular operation
163 // state and so will only be accessed by this start() method and the run()
164 // method.
165 std::lock_guard opLock{mut_};
166 thread_ = std::thread([this]() noexcept { this->run(); });
167
168 // Now that we've successfully launched the thread, increment the active
169 // thread count in the context. Do this before we release the lock so that
170 // we ensure the count increment happens before the count decrement that
171 // is performed when the thread is being retired.
172 ctx_->activeThreadCount_.fetch_add(1, std::memory_order_relaxed);
173 } UNIFEX_CATCH (...) {
174 unifex::set_error(std::move(receiver_), std::current_exception());
175 }
176}

178template <typename Receiver>
179inline void _op<Receiver>::type::run() noexcept {
180 // Read the thread_ and ctx_ members out from the operation-state
181 // and store them as local variables on the stack before calling the
182 // receiver completion-signalling methods as the receiver methods
183 // will likely end up destroying the operation-state object before
184 // they return.
185 context* ctx = ctx_;
186
187 std::thread thisThread;
188 {
189 // Wait until we can acquire the mutex here.
190 // This ensures that the read of thread_ happens-after the write to thread_
191 // inside start().
192 //
193 // TODO: This can be replaced with an atomic<bool>::wait() once we have
194 // access to C++20 atomics. This would eliminate the unnecessary synchronisation
195 // performed by the unlock() at end-of-scope here.
196 std::lock_guard opLock{mut_};
197 thisThread = std::move(thread_);
198 }
199
200 if (get_stop_token(receiver_).stop_requested()) {
201 unifex::set_done(std::move(receiver_));
202 } else {
203 UNIFEX_TRY {
204 unifex::set_value(std::move(receiver_));
205 } UNIFEX_CATCH (...) {
206 unifex::set_error(std::move(receiver_), std::current_exception());
207 }
208 }
209
210 ctx->retire_thread(std::move(thisThread));
211}
