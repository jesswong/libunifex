192 template <typename Value>
193 decltype(auto) await_transform(Value&& value) {
194 if constexpr (derived_from<remove_cvref_t<Value>, _task_base>) {
195 // We are co_await-ing a unifex::task, which completes inline because of task
196 // scheduler affinity. We don't need an additional transition.
197 return unifex::await_transform(*this, (Value&&) value);
198 } else if constexpr (tag_invocable<tag_t<unifex::await_transform>, type&, Value>
199 || detail::_awaitable<Value>) {
200 // Either await_transform has been customized or Value is an awaitable. Either
201 // way, we can dispatch to the await_transform CPO, then insert a transition back
202 // to the correct execution context if necessary.
203 return transform_awaitable_(unifex::await_transform(*this, (Value&&) value));
204 } else if constexpr (unifex::sender<Value>) {
205 return transform_sender_((Value&&) value);
206 } else {
207 // Otherwise, we don't know how to await this type. Just return it and let the
208 // compiler issue a diagnostic.
209 return (Value&&) value;
210 }
211 }
212
213 template <typename Awaitable>
214 decltype(auto) transform_awaitable_(Awaitable&& awaitable) {
215 if constexpr (blocking_kind::always_inline == cblocking<Awaitable>()) {
216 return Awaitable{(Awaitable&&) awaitable};
217 } else {
218 return unifex::await_transform(
219 *this,
220 finally(as_sender((Awaitable&&) awaitable), unstoppable(schedule(this->sched_))));
221 }
222 }
223
224 template <typename Sender>
225 decltype(auto) transform_sender_(Sender&& sndr) {
226 if constexpr (blocking_kind::always_inline == cblocking<Sender>()) {
227 return unifex::await_transform(*this, (Sender&&) sndr);
228 } else if constexpr (is_sender_for_v<remove_cvref_t<Sender>, schedule>) {
229 // If we are co_await'ing a sender that is the result of calling schedule,
230 // do something special
231 return transform_schedule_sender_((Sender&&) sndr);
232 } else {
233 // Otherwise, append a transition to the correct execution context and wrap the
234 // result in an awaiter:
235 return unifex::await_transform(*this,
236 finally((Sender&&) sndr, unstoppable(schedule(this->sched_))));
237 }
238 }
239
240 // co_await schedule(sched) is magical. It does the following:
241 // - transitions execution context
242 // - updates the coroutine's current scheduler
243 // - schedules an async cleanup action that transitions back to the correct
244 // context at the end of the coroutine (if one has not already been scheduled).
245 template <typename ScheduleSender>
246 decltype(auto) transform_schedule_sender_(ScheduleSender&& snd) {
247 // This sender is a scheduler provider. Get the scheduler. This get_scheduler
248 // call returns a reference to the scheduler stored within snd, which is an object
249 // whose lifetime spans a suspend point. So it's ok to build an any_scheduler_ref
250 // from it:
251 transform_schedule_sender_impl_(get_scheduler(snd));
252
253 // Return the inner sender, appropriately wrapped in an awaitable:
254 return unifex::await_transform(*this, std::move(snd).base());
255 }
256
257 decltype(auto) result() {
258 if (this->expected_.state_ == _state::exception) {
259 std::rethrow_exception(std::move(this->expected_.exception_).get());
260 }
261 return std::move(this->expected_.value_).get();
262 }
263 };

292struct _awaiter {

296 explicit type(coro::coroutine_handle<> coro) noexcept

299 // The move constructor is only ever called /before/ the awaitable is awaited.
300 // In those cases, the other fields have not been initialized yet and so do not
301 // need to be moved.

305 ~type() {

307 if constexpr (needs_stop_token_t::value)
308 stopTokenAdapter_.unsubscribe();
309 if constexpr (needs_scheduler_t::value)
310 sched_.destruct();
311 }
312 }

314 bool await_ready() noexcept {
315 return false;
316 }
317
318 coro::coroutine_handle<ThisPromise> await_suspend(
319 coro::coroutine_handle<OtherPromise> h) noexcept {
320 UNIFEX_ASSERT(coro_ && ((coro_ & 1u) == 0u));
321 auto thisCoro = coro::coroutine_handle<ThisPromise>::from_address((void*) coro_);
322 ++coro_; // mark the awaiter as needing cleanup
323 auto& promise = thisCoro.promise();
324 promise.continuation_ = h;
325 if constexpr (needs_scheduler_t::value) {
326 sched_.construct(get_scheduler(h.promise()));
327 promise.sched_ = sched_.get();
328 } else {
329 promise.sched_ = get_scheduler(h.promise());
330 }
331 if constexpr (needs_stop_token_t::value) {
332 promise.stoken_ = stopTokenAdapter_.subscribe(get_stop_token(h.promise()));
333 } else {
334 promise.stoken_ = get_stop_token(h.promise());
335 }
336 return thisCoro;
337 }
338
339 result_type await_resume() {
340 if constexpr (needs_stop_token_t::value)
341 stopTokenAdapter_.unsubscribe();
342 if constexpr (needs_scheduler_t::value)
343 sched_.destruct();
344 auto thisCoro = coro::coroutine_handle<ThisPromise>::from_address(
345 (void*) std::exchange(--coro_, 0));
346 coro_holder destroyOnExit{thisCoro};
347 return thisCoro.promise().result();
348 }

350 private:
351 using scheduler_t = remove_cvref_t<get_scheduler_result_t<OtherPromise&>>;
352 using stop_token_t = remove_cvref_t<stop_token_type_t<OtherPromise>>;
353 using needs_scheduler_t =
354 std::bool_constant<!same_as<scheduler_t, any_scheduler>>;
355 using needs_stop_token_t =
356 std::bool_constant<!same_as<stop_token_t, inplace_stop_token>>;
357
358 // Only store the scheduler and the stop_token in the awaiter if we need to type
359 // erase them. Otherwise, these members are "empty" and should take up no space

361 // Note: for the compiler to fold the members away, they must have different types.
