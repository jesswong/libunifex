59 // Clang before clang-12 has a bug with coroutines that self-destruct in an
60 // await_suspend that uses symmetric transfer. It appears that MSVC has the same
61 // bug, while Emscripten, the WebAssembly compiler just doesn't support tail calls yet.
62 // So instead of symmetric transfer, we accept the stack growth and resume
63 // the continuation from within await_suspend.
64#if (defined(__clang__) && (defined(__apple_build_version__) || __clang_major__ < 12)) || \
65 defined(_MSC_VER) || defined(__EMSCRIPTEN__)
66 template <typename CleanupPromise>
67 // Apple-clang and clang-10 and prior need for await_suspend to be noinline.
68 // MSVC and clang-11 can tolerate await_suspend to be inlined, so force it.
69#if defined(__apple_build_version__) || (defined(__clang__) && __clang_major__ <= 10)
70 UNIFEX_NO_INLINE
71#else
72 UNIFEX_ALWAYS_INLINE
73#endif
74 void await_suspend(coro::coroutine_handle<CleanupPromise> h) const noexcept {
75 auto continuation = h.promise().next();
76 h.destroy(); // The cleanup action has finished executing. Destroy it.
77 continuation.resume();
78 }
79#else
80 // No bugs here! OK to use symmetric transfer.
81 template <typename CleanupPromise>
82 coro::coroutine_handle<> await_suspend(coro::coroutine_handle<CleanupPromise> h) const noexcept {
83 auto continuation = h.promise().next();
84 h.destroy(); // The cleanup action has finished executing. Destroy it.
85 return continuation;
86 }
87#endif

89 void await_resume() const noexcept {
90 }

138// The die_on_done algorithm implemented here could be implemented in terms of
139// let_done, but this implementation is simpler since it doesn't instantiate
140// a bunch of templates that will never be needed (no need to connect or start a
141// sender returned from the done transform function).
142template <typename Receiver>
143struct _die_on_done_rec {
144 struct type {
145 Receiver rec_;
146 template (typename... Ts)
147 (requires receiver_of<Receiver, Ts...>)
148 void set_value(Ts&&... ts) && noexcept(is_nothrow_receiver_of_v<Receiver, Ts...>) {
149 unifex::set_value((Receiver&&) rec_, (Ts&&) ts...);
150 }
151 template (typename E)
152 (requires receiver<Receiver, E>)
153 void set_error(E&& e) && noexcept {
154 unifex::set_error((Receiver&&) rec_, (E&&) e);
155 }
156 [[noreturn]] void set_done() && noexcept {
157 UNIFEX_ASSERT(!"A cleanup action tried to cancel. Calling terminate...");
158 std::terminate();
159 }
160
161 template(typename CPO)
162 (requires is_receiver_query_cpo_v<CPO> AND
163 is_callable_v<CPO, const Receiver&>)
164 friend auto tag_invoke(CPO cpo, const type& p)
165 noexcept(is_nothrow_callable_v<CPO, const Receiver&>)
166 -> callable_result_t<CPO, const Receiver&> {
167 return cpo(p.rec_);
168 }
169 };
